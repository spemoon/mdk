lang.timer

说明

定时器加强

用法 
                	// 声明定时器:
		            var timer = lang.timer({
		                fn:function() {
		                    // 每个节拍运行的代码
		                },
		                rule:function() {
		                    // 运行每个节拍前的规则，true则继续，false则停止
		                },
		                step:function() {
		                    // 节拍间隔，可选，默认128ms，可以是一个毫秒级别常数，也可以是个function返回毫秒数达到不等间隔节拍
		                },
		                start: function() {
						    // 节拍开始前的callback
						},
		                end: function() {
                            // 节拍结束后的callback
                        }
		            });
					
					// 终止定时器：
					timer.stop();
